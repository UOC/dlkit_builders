    ##
    # Implemented via osid.calendaring.DateTime init.txt template

    ##
    # The default CalendarType identifier triple for the Python datetime 
    # module should probably come from a config file or something. But 
    # here it is for now:
    _defaultCalendarAuthority = 'python.org'
    _defaultCalendarNamespace = 'datetime.date'
    _defaultCalendarIdentifier = 'PYDATE'
    
    ##
    # These are being explicitly declared  as attribute to make sure
    # none of them fall through to __getattr__.
    global _aeon
    global _epoch
    global _millennium
    global _century
    global _year
    global _month
    global _day
    global _definesUncertainty
    global _uncertaintyUnits
    global _uncertaintyMinus
    global _uncertaintyPlus

    def __init__(self, pythonDateTimeObject = None,
                       calendarType = None,
                       aeon = 0,
                       epoch = 0,
                       millennium = 0,
                       century = 0,
                       year = 0,
                       month = 0,
                       day = 0,
                       definesUncertainty = None,
                       uncertaintyUnits = None,
                       uncertaintyMinus = None,
                       uncertaintyPlus = None,
                       *args):

        import datetime
        if pythonDateTimeObject:
            Time.__init__(self, pythonDateTimeObject, *args)
            self._day = pythonDateTimeObject.day
            self._month = pythonDateTimeObject.month
            self._year = pythonDateTimeObject.year
        else:
            self._day = day
            self._month = month
            self._year = year
        self._century = century
        self._millennium = millennium
        self._epoch = epoch
        self._aeon = aeon
        self._definesUncertainty = definesUncertainty,
        self._uncertaintyUnits = uncertaintyUnits
        self._uncertaintyMinus = uncertaintyMinus
        self._uncertaintyPlus = uncertaintyPlus
        
        if calendarType:
            self._calendarType = calendarType
        else:
            if pythonDateTimeObject:
                self._calendarType = self._get_type(self._defaultCalendarAuthority,
                                                    self._defaultCalendarNamespace,
                                                    self._defaultCalendarIdentifier)
            else:
                self._calendarType = self._get_type('unknown',
                                                    'unknown',
                                                    'unknown')
